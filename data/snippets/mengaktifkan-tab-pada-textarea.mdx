Ini dimaksudkan untuk mengaktifkan tab pada textarea.

```js:enableTab.js showLineNumbers
function enableTab(el) {
  el.onkeydown = function (e) {
    if (e.keyCode === 9) {
      var val = this.value,
        start = this.selectionStart,
        end = this.selectionEnd
      this.value = val.substring(0, start) + '\t' + val.substring(end)
      this.selectionStart = this.selectionEnd = start + 1
      return false
    }
  }
}
```

## Penjelasan

Berikut cuplikan yang sama, dengan komentar sebaris yang menawarkan informasi tambahan:

```js:enableTab.js showLineNumbers
function enableTab(el) {
  el.onkeydown = function (e) {
    if (e.keyCode === 9) {
      // tab ditekan

      // dapatkan posisi/seleksi tanda sisipan
      var val = this.value,
        start = this.selectionStart,
        end = this.selectionEnd

      // atur nilai textarea ke: teks sebelum tanda sisipan + tab + teks setelah tanda sisipan
      this.value = val.substring(0, start) + '\t' + val.substring(end)

      // letakkan tanda sisipan pada posisi yang benar lagi
      this.selectionStart = this.selectionEnd = start + 1

      // mencegah kehilangan fokus
      return false
    }
  }
}
```

## Pengunaan

```js:index.js showLineNumbers {2}
const textarea = document.querySelector('textarea')
enableTab(textarea)
```
