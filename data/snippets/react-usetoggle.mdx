Ini berguna ketika kita ingin melakukan tindakan yang berlawanan, misalnya: tampilkan dan sembunyikan modal, tampilkan lebih banyak/tampilkan lebih sedikit teks, buka/tutup menu samping.

## Versi Javascript

```js:useToggle.js showLineNumbers {7}
import { useCallback, useState } from 'react';

// Penggunaan
function App() {

     // Panggil hook yang mengembalikan, nilai saat ini dan fungsi toggler
    const [isTextChanged, setIsTextChanged] = useToggle();

    return (
        <button onClick={setIsTextChanged}>{isTextChanged ? 'Toggled' : 'Click to Toggle'}</button>
    );
}

// Hook
// Parameter adalah boolean, dengan nilai default "false"
const useToggle = (initialState = false) => {

    // Inisialisasi status
    const [state, setState] = useState(initialState);

     // Mendefinisikan dan menghafal fungsi toggler jika kita mewariskan komponen,
    // Fungsi ini mengubah nilai boolean menjadi nilai kebalikannya
    const toggle = useCallback(() => setState(state => !state), []);

    return [state, toggle]
}

```

## Versi Typescript

```ts:useToggle.tsx showLineNumbers {6}
import { useCallback, useState } from 'react';

// Penggunaan
function App() {
    // Panggil hook yang mengembalikan, nilai saat ini dan fungsi toggler
    const [isTextChanged, setIsTextChanged] = useToggle();
    return (
        <button onClick={setIsTextChanged}>{isTextChanged ? 'Toggled' : 'Click to Toggle'}</button>
    );
}

// Hook
// Parameter adalah boolean, dengan nilai default "false"
const useToggle = (initialState: boolean = false): [boolean, any] => {

    // Inisialisasi status
    const [state, setState] = useState<boolean>(initialState);

    // Mendefinisikan dan menghafal fungsi toggler jika kita mewariskan komponen,
    // Fungsi ini mengubah nilai boolean menjadi nilai kebalikannya
    const toggle = useCallback((): void => setState(state => !state), []);

    return [state, toggle]
}

```
